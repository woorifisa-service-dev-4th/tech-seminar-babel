ES5 코드 실행

ES5 문법을 사용한 코드입니다. 모든 브라우저에서 정상적으로 실행됩니다. 


function Person(name, age) {
    this.name = name;
    this.age = age;
}

Person.prototype.greet = function() {
    return "안녕하세요, 저는 " + this.name + "입니다. 나이는 " + this.age + "살이에요.";
};

var person = new Person("홍길동", 25);
document.getElementById("output").innerText = person.greet();
